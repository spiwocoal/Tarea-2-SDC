    \begin{itemize}
        \item \(Z_0 = 0.4062\)
        \item \(Z_1 = -1.0472\)
    \end{itemize}
  \end{multicols}

  \item \textbf{L.C} =  \(\frac{  \mora{k_{c}} \nara{ k_{a}} z^{-1} \cdot \frac{\rojo{\psi}(z)}{\verd{v_i}(z)}} {1+ \mora{k_{c}} \nara{ k_{a}} z^{-1} \cdot \frac{\rojo{\psi}(z)}{\verd{v_i}(z)} \nara{k_{st}} }  \)

 \end{itemize}
